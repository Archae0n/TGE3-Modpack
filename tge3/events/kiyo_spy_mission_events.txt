country_event = {
	id = kiyo_event_spy.13
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		remove_country_flag = kiyo_spy_scandal
		
		if = { limit = { has_country_flag = kiyo_spy_scandal_victim_a }
			if = { limit = { NOT = { any_country = { has_relation_flag = { who = root flag = kiyo_spy_scandal_a } } } }
				add_modifier = { modifier = kiyo_spy_scandal_trial_good days = 1800 }
				country_event = { id = kiyo_event_spy.14 }
			}
			else = {
				every_country = { limit = { has_relation_flag = { who = root flag = kiyo_spy_scandal_a } }
					add_resource = { influence = 50 }
					kiyo_spy_trigger_increase_intrigue = yes
				}
				add_modifier = { modifier = kiyo_spy_scandal_bad days = 1800 }
				if = {
					limit = {
						any_country = {
							OR = {
								has_relation_flag = { who = root flag = kiyo_spy_intrigue3 }
								has_relation_flag = { who = root flag = kiyo_spy_intrigue4 }
								has_relation_flag = { who = root flag = kiyo_spy_intrigue5 }
							}
						}
					}
					add_modifier = { modifier = kiyo_spy_scandal_bad_intrigue_3 days = 3600 }
				}
				country_event = { id = kiyo_event_spy.15 }
			}
		}
		else_if = { limit = { has_country_flag = kiyo_spy_scandal_victim_b }
			if = { limit = { NOT = { any_country = { has_relation_flag = { who = root flag = kiyo_spy_scandal_b } } } }
				event_target:kiyo_spy_scandal_leader = { add_experience = 200 }
				country_event = { id = kiyo_event_spy.16 }

				every_country = { limit = { has_relation_flag = { who = root flag = kiyo_spy_scandal_b } }
					add_resource = { influence = 50 }
					kiyo_spy_trigger_increase_intrigue = yes
				}
				add_modifier = { modifier = kiyo_spy_scandal_bad days = 1800 }
				if = {
					limit = {
						any_country = {
							OR = {
								has_relation_flag = { who = root flag = kiyo_spy_intrigue3 }
								has_relation_flag = { who = root flag = kiyo_spy_intrigue4 }
								has_relation_flag = { who = root flag = kiyo_spy_intrigue5 }
							}
						}
					}
					add_modifier = { modifier = kiyo_spy_scandal_bad_intrigue_3 days = 3600 }
				}
				country_event = { id = kiyo_event_spy.17 }
			}
		}
		else = {
			every_country = { limit = { has_relation_flag = { who = root flag = kiyo_spy_scandal_a } }
				add_resource = { influence = 50 }
				kiyo_spy_trigger_increase_intrigue = yes
			}
			every_country = { limit = { has_relation_flag = { who = root flag = kiyo_spy_scandal_b } }
				add_resource = { influence = 50 }
				kiyo_spy_trigger_increase_intrigue = yes
			}
			add_modifier = { modifier = kiyo_spy_scandal_bad days = 1800 }
			if = {
				limit = {
					any_country = {
						OR = {
							has_relation_flag = { who = root flag = kiyo_spy_intrigue3 }
							has_relation_flag = { who = root flag = kiyo_spy_intrigue4 }
							has_relation_flag = { who = root flag = kiyo_spy_intrigue5 }
						}
					}
				}
				add_modifier = { modifier = kiyo_spy_scandal_bad_intrigue_3 days = 3600 }
			}
			log = "HUMAN ERROR in kiyo_event_spy.13: The spying empire [Root.GetName] apparently did not make a choice..."
		}
		
		every_country = {
			limit = {
				has_relation_flag = { who = root flag = kiyo_spy_spying_on }
				OR = {
					has_relation_flag = { who = root flag = kiyo_spy_scandal_a }
					has_relation_flag = { who = root flag = kiyo_spy_scandal_b }
				}
			}
			remove_relation_flag = { who = root flag = kiyo_spy_spying_on }
			remove_relation_flag = { who = root flag = kiyo_spy_scandal_a }
			remove_relation_flag = { who = root flag = kiyo_spy_scandal_b }
			set_timed_relation_flag = { who = root flag = kiyo_spy_spying_cooldown days = 3600 }
			country_event = { id = kiyo_event_notification.0 days = 3601 }
		}
		remove_country_flag = kiyo_spy_scandal_victim_a
		remove_country_flag = kiyo_spy_scandal_victim_b
